def count(seq):
    """
    Counts the number of elements in a given list
    including elements in inner lists
    """
    if not seq:
        return 0
    elif isinstance(seq[0], list):
        return count(seq[0]) + count(seq[1:])
    else:
        return 1 + count(seq[1:])
    \end{lstlisting}
